integration_of_territories = {

	settle_state = {
		state_target = yes
		on_map_mode = map_and_decisions_view

		target_trigger = {
			FROM = {
				is_owned_by = ROOT
				has_state_flag = uninhabited_state
				impassable = no
			}
		}

		visible = {
			any_state = {
				is_owned_by = ROOT
				has_state_flag = uninhabited_state
				impassable = no
			}
		}

		available = {
			capital_scope = {
				state_population > 2500
			}
		}
		
		remove_effect = {
			capital_scope = {
				add_manpower = -2500
			}
			FROM = {
				if = {
					limit = {
						infrastructure < 1
					}
					add_building_construction = {
						type = infrastructure
						level = 1
						instant_build = yes
					}
				}
				add_manpower = 2500
				clr_state_flag = uninhabited_state
			}
		}
		
		icon = infiltrate_state
		
		days_remove = 100

		cost = 100
		
		ai_will_do = {
			factor = 1
		}
	}

	integrate_state = {
		state_target = yes
		on_map_mode = map_and_decisions_view

		target_trigger = {
			FROM = {
				is_owned_by = ROOT
				NOT = { is_core_of = ROOT }
				NOT = { has_state_flag = uninhabited_state }
			}
		}

		visible = {
			any_state = {
				is_owned_by = ROOT
				NOT = { is_core_of = ROOT }
				NOT = { has_state_flag = uninhabited_state }
			}
		}
		
		available = {
			FROM = {
				OR = {
					has_active_resistance = no
					compliance > 80
				}
			}
		}
		
		remove_effect = {
			FROM = {
				add_core_of = ROOT
			}
		}
		
		icon = infiltrate_state
		
		days_remove = 365

		cost = 100
		
		ai_will_do = {
			factor = 1
		}
	}

}
