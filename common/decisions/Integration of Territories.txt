			}

			}
		}

			FROM = {
				OR = {
					has_active_resistance = no
					compliance > 80
				}
			}
		}
		
		remove_effect = {
			FROM = {
				add_core_of = ROOT
			}
		}
		
		icon = infiltrate_state
		
		cost = 100
		
		ai_will_do = {
			factor = 1
		}
	}
