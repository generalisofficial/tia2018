scripted_diplomatic_actions = {

	recall_volunteers = {
		allowed = {
			ROOT = { is_ai = no }
		}

		visible = {
			always = yes
		}

		selectable = {
			if = {
				limit = {
					NOT = { has_volunteers_amount_from = { tag = ROOT count > 0 } }
				}
				custom_trigger_tooltip = {
					tooltip = RECALL_VOLUNTEERS_TOOLTIP
					always = no
				}
			}
			else = {
				custom_trigger_tooltip = {
					tooltip = RECALL_VOLUNTEERS_TOOLTIP_G
					always = yes
				}
			}
		}

		cost = 0
		command_power = 0

		show_acceptance_on_action_button = no

		icon = 1

		complete_effect = {
			ROOT = { recall_volunteers_from = PREV }
		}

		ai_acceptance = {
			condition = {
				base = 100
			}
		}
	}

	trade_embargo = {
		allowed = {
			ROOT = {
				is_sovereign_state = yes
				NOT = {
					has_opinion_modifier = {
						target = FROM
						modifier = trade_embargo_trade
					}
				}
			}
		}

		visible = {
			is_sovereign_state = yes
			NOT = { 
				OR = {
					is_puppet_of = ROOT
					ROOT = { is_puppet_of = FROM }
					ROOT = {
						has_opinion_modifier = {
							target = FROM
							modifier = trade_embargo_trade
						}
					}
				}
			}
		}

		can_be_accepted = {
			always = yes
		}

		can_be_sent = {
			always = yes
		}

		send_description = TRADE_EMBARGO_LOC
		accept_title = TRADE_EMBARGO_INITIATED_TITLE
		accept_description = TRADE_EMBARGO_INITIATED_DESC
		
		cost = 50

		requires_acceptance = no
		
		complete_effect = {
			add_opinion_modifier = {
    			target = FROM
    			modifier = trade_embargo_trade
			}
			FROM = {
				add_opinion_modifier = {
    				target = ROOT
    				modifier = trade_embargo
				}
				add_opinion_modifier = {
    				target = ROOT
    				modifier = trade_embargo_trade
				}
			}
		}
	}

	cancel_trade_embargo = {
		allowed = {	
			always = yes
		}

		visible = {
			has_opinion_modifier = {
    			target = FROM
    			modifier = trade_embargo_trade
			}
			FROM = {
				has_opinion_modifier = {
					target = ROOT
					modifier = trade_embargo
				}
			}
		}

		can_be_accepted = {
			always = yes
		}

		can_be_sent = {
			always = yes
		}

		send_description = CANCEL_TRADE_EMBARGO_LOC
		accept_title = CANCEL_EMBARGO_INITIATED_TITLE
		accept_description = CANCEL_TRADE_EMBARGO_INITIATED_DESC
		
		cost = 100

		requires_acceptance = no

		complete_effect = {
			remove_opinion_modifier = {
    			target = FROM
    			modifier = trade_embargo_trade
			}
			FROM = {
				remove_opinion_modifier = {
    				target = ROOT
    				modifier = trade_embargo
				}
				remove_opinion_modifier = {
    				target = ROOT
    				modifier = trade_embargo_trade
				}
			}
		}
	}
}
