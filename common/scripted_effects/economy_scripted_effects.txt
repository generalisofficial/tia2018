economy_setup = {
	update_finances = yes
}

update_finances = {
	set_variable = { monthly_tax_revenue = 0 }
	set_variable = { monthly_expenses = 0 }

	# INCOME - TAXES
	if = {
		limit = {
			has_idea = very_high_taxes
		}
		#
	}
	if = {
		limit = {
			has_idea = high_taxes
		}
		#
	}
	if = {
		limit = {
			has_idea = moderate_taxes
		}
		#
	}
	if = {
		limit = {
			has_idea = low_taxes
		}
		#
	}
	if = {
		limit = {
			has_idea = very_low_taxes
		}
		#
	}

	# EXPENSES

	set_variable = { monthly_net_income  = monthly_tax_revenue }
	subtract_from_variable = {
		var = monthly_net_income
		value = monthly_expenses
	}
}

monthly_economy_pulse = {
	update_finances = yes
	if = {
		limit = {
			check_variable = {
				var = monthly_expenses
				value = 0
				compare = less_than
			}
		}
		subtract_from_variable = {
			var = state_treasury
			value = monthly_net_income
		}
	}
	if = {
		limit = {
			check_variable = {
				var = monthly_expenses
				value = 0
				compare = greater_than
			}
		}
		add_to_variable = {
			var = state_treasury
			value = monthly_net_income
		}
	}
}
