should_initiate_resistance = {
    state_population > 1
    OR = {
        AND = {
            check_variable = { core_countries^num > 0 }
            NOT = {
                is_owned_by = FROM
                is_core_of = FROM  
            }
        }
        AND = {
            check_variable = { core_countries^num > 1 }
            FROM = { is_democracy = no }
        }
    }
}
