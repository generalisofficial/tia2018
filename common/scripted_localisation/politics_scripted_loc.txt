####################### GET IF COALITION FORMING IS FINISHED

defined_text = {
	name = GetCoalitionFinished

	text = {
		trigger = {
			has_at_least_one_government_and_one_coalition_flag = yes
		}
		localization_key = COALITION_FINISHED_OK
	}

	text = {
		trigger = {
			NOT = { has_at_least_one_government_and_one_coalition_flag = yes }
			check_variable = {
				var = ROOT.NewCoalition
				value = 50
				compare = greater_than_or_equals
			}
		}
		localization_key = COALITION_FINISHED_NO
	}
}
