# Get Country Type (for diplo menu)
defined_text = {
	name = GetCountryType
	text = {
		trigger = {
			has_idea = violent_non_state_actor
		}
		localization_key = violent_non_state_actor
	}
	text = {
		trigger = {
			has_autonomy_state = autonomy_autonomous_region
		}
		localization_key = CT_devolved
	}
	text = {
		trigger = {
			has_idea = federal
			NOT = { has_autonomy_state = autonomy_autonomous_region }
		}
		localization_key = CT_federal
	}
	text = {
		trigger = {
			has_idea = unitary
			NOT = { has_autonomy_state = autonomy_autonomous_region }
		}
		localization_key = CT_unitary
	}
}

defined_text = {
	name = GetFormofGovernment
	text = {
		trigger = {
			has_idea = parliamentary_system
			OR = {
				has_idea = idea_constitutional_monarchy
				has_idea = idea_commonwealth_realm
			}
		}
		localization_key = FoG_parliamentary_constitutional_monarchy
	}
	text = {
		trigger = {
			has_idea = parliamentary_system
			NOT = { 			
				OR = {
					has_idea = idea_constitutional_monarchy
					has_idea = idea_commonwealth_realm
				} 
			}
		}
		localization_key = FoG_parliamentary_republic
	}
	text = {
		trigger = {
			has_idea = absolute_monarchy
		}
		localization_key = FoG_absolute_monarchy
	}
	text = {
		trigger = {
			has_idea = one_party_state
			NOT = { 
				OR = {
					has_government = communist 
					has_country_leader_ideology = khomeinist
				}
			}
		}
		localization_key = FoG_one_party_presidential_republic
	}
	text = {
		trigger = {
			has_idea = one_party_state
			has_government = communist
		}
		localization_key = FoG_one_party_socialist_republic
	}
	text = {
		trigger = {
			has_idea = one_party_state
			has_country_leader_ideology = khomeinist
		}
		localization_key = FoG_one_party_islamic_republic
	}
	text = {
		trigger = {
			OR = {
				has_idea = presidential_republic
				has_idea = demarchic_republic
			}
		}
		localization_key = FoG_presidential_republic
	}
}
