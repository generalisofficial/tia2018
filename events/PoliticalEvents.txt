#Fascist Ultimatum
country_event = {
	id = political.1
	title = political.1.t
	desc = political.1.d
	
	trigger = {
		fascist > 0.5
		NOT = { has_government = fascist has_elections = yes }
	}
	
	mean_time_to_happen = {
		days = 100
		modifier = { factor = 0.5 fascist > 0.48 }
		modifier = { factor = 0.5 fascist > 0.52 }
		modifier = { factor = 0.5 fascist > 0.56 }
		modifier = {
			factor = 0.75
			has_government = communist
		}
	}
	
	option = {
		name = political.1.o1
		set_politics = {
			ruling_party = fascist
			elections_allowed = no
		}
		news_event = {
			hours = 3
			id = news.1
		}
		break_coalition = yes
		quickset_coalition = yes
	}
	
	option = {
		name = political.1.o1
		start_civil_war = {
			ideology = fascist
			size = 0.5
		}
	}
}
