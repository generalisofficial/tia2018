#New UNSC members
news_event = {
	id = news.53
	title = news.53.t
	desc = news.53.d
	picture = GFX_news_event_united_nations
	
	is_triggered_only = yes
	major = yes
	
	option = {
		name = news.53.o1
		trigger = {
			has_idea = idea_united_nations_security_council_member
		}
	}
	option = {
		name = news.53.o2
		trigger = {
			NOT = { has_idea = idea_united_nations_security_council_member }
			has_idea = idea_united_nations_NP_security_council_member
		}
	}
	option = {
		name = news.53.o3
		trigger = {
			NOT = { has_idea = idea_united_nations_security_council_member }
			NOT = { has_idea = idea_united_nations_NP_security_council_member }
			NOT = { has_idea = non_member_state }
			NOT = { has_idea = salafist_jihadist_group }
		}
	}
	option = {
		name = news.53.o4
		trigger = {
			NOT = { has_idea = idea_united_nations_security_council_member }
			NOT = { has_idea = idea_united_nations_NP_security_council_member }
			OR = {
				has_idea = non_member_state
				has_idea = salafist_jihadist_group
			}
		}
	}
}

		name = news.80.o4

			NOT = { tag = ROOT }

		}
	}
