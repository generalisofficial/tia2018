		multiply_variable = { monarchist_seats = 100 }
		set_variable = { right_wing_nationalist_seats = party_popularity@right_wing_nationalist }
		multiply_variable = { right_wing_nationalist_seats = 100 }
		set_variable = { fascist_seats = party_popularity@fascist }
		multiply_variable = { fascist_seats = 100 }
		set_variable = { islamist_seats = party_popularity@islamist }
		multiply_variable = { islamist_seats = 100 }
		set_variable = { ROOT.Coalition = party_popularity@ruling_party }
		multiply_variable = { ROOT.Coalition = 100 }
		set_parliament_graphics = yes
	}
	
	option = {
		name = election.2.o1
		custom_effect_tooltip = TT_parliamentary_election_results
		if = {
			limit = {
				minority_government = yes
			}
			add_ideas = minority_government
		}
    }
}

#Election - Parliamentary
country_event = {
	id = election.3
	title = election.3.t
	desc = election.3.d
	
	is_triggered_only = yes
	
	trigger = {
	    has_idea = parliamentary_republic
	}
	
	immediate = {
	    break_coalition = yes
	}
	
	option = {
		name = election.3.o1
		custom_effect_tooltip = TT_parliamentary_election_results
		quickset_parliament = yes
		if = {
			limit = {
			    OR = {
					communist > 0.5
					socialist > 0.5
					progressive > 0.5
					social_democrat > 0.5
					social_liberal > 0.5
					classical_liberal > 0.5
					conservative > 0.5
					national_conservative > 0.5
					monarchist > 0.5
					right_wing_nationalist > 0.5
					fascist > 0.5
					islamist > 0.5
				}
			}
			country_event = election.4
			else = {
				set_country_flag = forming_government_1
			}
		}
    }
}
