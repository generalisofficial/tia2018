add_namespace = coup

#Islamist Coup
country_event = {
	id = coup.1
	title = coup.1.t
	desc = coup.1.d
	
	trigger = {
		islamist > 0.5
		NOT = { has_government = islamist }
		is_democracy = no
        has_civil_war = no
		NOT = { has_idea = violent_non_state_actor }
	}
	
	mean_time_to_happen = {
		days = 100
		modifier = {
			factor = 0.5
			islamist > 0.5
		}
		modifier = {
			factor = 0.5
			islamist > 0.55
		}
		modifier = {
			factor = 1.5
			communist > 0.3
		}
	}
	
	option = {
		name = coup.o1
		set_politics = {
			ruling_party = islamist
			elections_allowed = no
		}
		add_popularity = { 
			ideology = islamist
			popularity = 0.05
		}
		news_event = coup.13
		one_party_parliament = yes
		break_coalition = yes
		quickset_coalition = yes
	}
}
