add_namespace = venezuela

country_event = { #Juan Guaido declared president
	id = venezuela.1
	title = venezuela.1.t
	desc = venezuela.1.d

	fire_only_once = yes

	trigger = {
		original_tag = VEN
		date > 2019.1.23
		check_variable = {
        	var = social_liberal_seats
        	value = 50
        	compare = greater_than_or_equals
    	}
		has_country_leader = {
			name = "Nicolás Maduro"
			ruling_only = yes
		}
	}
	
	option = {
		name = venezuela.1.o1
		add_ideas = idea_venezuelan_presidential_crisis
		news_event = venezuela.2
	}
}

country_event = { #Operación Libertad
	id = venezuela.3
	title = venezuela.3.t
	desc = venezuela.3.d

	fire_only_once = yes

	trigger = {
		original_tag = VEN
		date > 2019.4.30
		has_idea = idea_venezuelan_presidential_crisis
	}
	
	option = {
		name = venezuela.1.o1
		random_list = {
			80 = {
				news_event = venezuela.4
				hidden_effect = {
					remove_ideas = idea_venezuelan_presidential_crisis
					set_politics = {	
						ruling_party = social_liberal
						elections_allowed = yes
					}
					retire_ideology_leader = socialist
					create_country_leader = {
						name = "Diosdado Cabello"
						picture = GFX_Portrait_VEN_Diosdado_Cabello
						ideology = socialism_of_the_21st_century
					}
					set_variable = { communist_seats = 0 }
					set_variable = { socialist_seats = 0 }
					set_variable = { progressive_seats = 0 }
					set_variable = { social_democrat_seats = 0 }
					set_variable = { social_liberal_seats = 100 }
					set_variable = { classical_liberal_seats = 0 }
					set_variable = { conservative_seats = 0 }
					set_variable = { national_conservative_seats = 0 }
					set_variable = { monarchist_seats = 0 }
					set_variable = { right_wing_populist_seats = 0 }
					set_variable = { fascist_seats = 0 }
					set_variable = { islamist_seats = 0 }
					set_variable = { ROOT.Coalition = 100 }
					set_variable = { ROOT.Opposition = 0 }
					set_parliament_graphics = yes
				}
			}
			20 = {
			  	news_event = venezuela.5
				hidden_effect = {
					remove_ideas =  idea_venezuelan_presidential_crisis
				}
			}
		}
	}
}
