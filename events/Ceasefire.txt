country_event = { #(FROM) accepts ceasefire
	id = ceasefire.2
	title = ceasefire.2.t
	desc = ceasefire.2.d
	
	is_triggered_only = yes
	
	option = {
		name = ceasefire.2.o1
		FROM = { save_event_target_as = ceasefire_receiver } 
		news_event = ceasefire.4
		add_political_power = 150
		every_subject_country = {
			every_state = {
				limit = {
					is_owned_by = FROM
					OR = {
						is_controlled_by = ROOT
						is_controlled_by = PREV
					}
					OR = {
						is_core_of = PREV
						is_claimed_by = PREV
					}
				}
				transfer_state_to = PREV
			}
		}
		FROM = { 
			every_subject_country = {
				every_state = {
					limit = {
						is_owned_by = ROOT
						OR = {
							is_controlled_by = FROM
							is_controlled_by = PREV
						}
						OR = {
							is_core_of = PREV
							is_claimed_by = PREV
						}
					}
					transfer_state_to = PREV
				}
			}
		}
		every_state = {
			limit = {
				is_owned_by = FROM
				is_controlled_by = ROOT
			}
			transfer_state_to = ROOT
		}
		every_state = {
			limit = {
				is_owned_by = ROOT
				is_controlled_by = FROM
			}
			transfer_state_to = FROM
		}
		white_peace = FROM
	}
}
