country_event = { #Sargsyan for Prime Minister?
	id = armenia.1
	title = armenia.1.t
	desc = armenia.1.d
	
	fire_only_once = yes

	trigger = {
		original_tag = ARM
		has_government = national_conservative
		has_idea = parliamentary_system
		NOT = { has_country_flag = coalition_social_liberal }
		date > 2018.4.9
		date < 2019.1.1
	}
	
	option = {
		name = armenia.1.o1
		ai_chance = {
			factor = 60
		}
		retire_country_leader = yes
		create_country_leader = {
	        name = "Serzh Sargsyan"
	        picture = GFX_Portrait_ARM_Serzh_Sargsyan
	        ideology = national_conservative_ideology
			traits = { strongman friend_of_russia }
        }
		news_event = armenia.2
		add_ideas = idea_large_scale_protests
		set_country_flag = velvet_revolution
		add_popularity = {
			ideology = social_liberal
			popularity = 0.1
		}
	}

	option = {
		name = armenia.1.o2
		ai_chance = {
			factor = 40
		}
	}
}
